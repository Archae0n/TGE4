			set_variable = {

			}
			#1	
			while = {
				count = 99
				if = {
					limit = {
						resource_stockpile_compare = { resource = energy value >= 10000 }

					add_resource = { energy = -10000 }

					limit = {

						change_variable = {
							which = ftm_buyer_marketed_fleet_slot_1_check_energy
							value = 100
						}
					}
				}
				else_if = {
					limit = {
						resource_stockpile_compare = { resource = energy value >= 10 }
					}
					add_resource = { energy = -10 }
					event_target:ftm_marketed_fleet_slot_1_seller = { 
						change_variable = {
							which = ftm_buyer_marketed_fleet_slot_1_check_energy
							value = 10
						}		
					}
				}
				else_if = {
					limit = {
						resource_stockpile_compare = { resource = energy value >= 1 }
					}
					add_resource = { energy = -1 }
					event_target:ftm_marketed_fleet_slot_1_seller = { 
						change_variable = {
							which = ftm_buyer_marketed_fleet_slot_1_check_energy
							value = 1
						}
					}					
				}
				else = {
					#nothing
				}
			}
			#2
			event_target:ftm_marketed_fleet_slot_1_seller = { 
				set_variable = {
					which = ftm_buyer_marketed_fleet_slot_1_energy_current_count
					value = ftm_buyer_marketed_fleet_slot_1_check_energy
				}
			}
			#3
			while = {
				count = 99
				if = {
					limit = {
						event_target:ftm_marketed_fleet_slot_1_seller = { 
							check_variable = {
								which = ftm_buyer_marketed_fleet_slot_1_check_energy
								value >= 10000
							}
						}
					}
					add_resource = { energy = 10000 }
					event_target:ftm_marketed_fleet_slot_1_seller = { 
						change_variable = {
							which = ftm_buyer_marketed_fleet_slot_1_check_energy
							value = -10000
						}

					limit = {
						event_target:ftm_marketed_fleet_slot_1_seller = {

						event_target:ftm_marketed_fleet_slot_1_seller = { 
							check_variable = {
								which = ftm_buyer_marketed_fleet_slot_1_check_energy

							}

						change_variable = {
							which = ftm_buyer_marketed_fleet_slot_1_check_energy
							value = -100
						}
					}					
				}
				else_if = {
					limit = {
						event_target:ftm_marketed_fleet_slot_1_seller = { 
							check_variable = {
								which = ftm_buyer_marketed_fleet_slot_1_check_energy

						change_variable = {

					#nothing

			log = "[Root.GetName]: energy_check complete. ENERGY: [event_target:ftm_marketed_fleet_slot_1_seller.ftm_buyer_marketed_fleet_slot_1_energy_current_count]."
			
			
			### 
			
			#4
			if = {
				limit = {
					event_target:ftm_marketed_fleet_slot_1_seller = { 
						check_variable = {
							which = ftm_buyer_marketed_fleet_slot_1_energy_current_count

					}

					while = {
						count = 99
						if = {
							limit = {
								event_target:ftm_marketed_fleet_slot_1_seller = { 
									check_variable = {
										which = ftm_marketed_fleet_slot_1_lot_cost
										value >= 10000
									}
								}
							}
							add_resource = { energy = -10000 }
							event_target:ftm_marketed_fleet_slot_1_seller = { 
								change_variable = {
									which = ftm_marketed_fleet_slot_1_lot_cost
									value = -10000
								}
							}
						}
						else_if = {
							limit = {
								event_target:ftm_marketed_fleet_slot_1_seller = { 
									check_variable = {
										which = ftm_marketed_fleet_slot_1_lot_cost
										value >= 1000
									}
								}
							}
							add_resource = { energy = -1000 }
							event_target:ftm_marketed_fleet_slot_1_seller = { 
								change_variable = {
									which = ftm_marketed_fleet_slot_1_lot_cost
									value = -1000
								}		
							}
						}
						else_if = {
							limit = {
								event_target:ftm_marketed_fleet_slot_1_seller = { 
									check_variable = {
										which = ftm_marketed_fleet_slot_1_lot_cost
										value >= 100
									}
								}
							}
							add_resource = { energy = -100 }
							event_target:ftm_marketed_fleet_slot_1_seller = { 
								change_variable = {
									which = ftm_marketed_fleet_slot_1_lot_cost
									value = -100
								}
							}					
						}
						else_if = {
							limit = {
								event_target:ftm_marketed_fleet_slot_1_seller = { 
									check_variable = {
										which = ftm_marketed_fleet_slot_1_lot_cost
										value >= 10
									}
								}
							}
							add_resource = { energy = -10 }
							event_target:ftm_marketed_fleet_slot_1_seller = { 
								change_variable = {
									which = ftm_marketed_fleet_slot_1_lot_cost
									value = -10
								}
							}
						}
						else_if = {
							limit = {
								event_target:ftm_marketed_fleet_slot_1_seller = { 
									check_variable = {
										which = ftm_marketed_fleet_slot_1_lot_cost
										value >= 1
									}
								}
							}
							add_resource = { energy = -1 }
							event_target:ftm_marketed_fleet_slot_1_seller = { 
								change_variable = {
									which = ftm_marketed_fleet_slot_1_lot_cost
									value = -1
								}					
							}
						}
						else = {
							#nothing
						}
					}	
				}
				event_target:ftm_marketed_fleet_slot_1_seller = { 
					country_event = { id = ftm_market.112 }
				}
				remove_global_flag = ftm_market_slot_1_selected	
			}
		}
	}
}
