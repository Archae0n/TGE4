		is_domestic_robot = yes
		is_domestic_servant = yes
		is_slave_complex_possible = yes
	}

	is_enslaved = yes

	NOT = { has_slavery_type = { type = slavery_livestock } }
	NOT = { has_slavery_type = { type = slavery_matrix } }	
}
