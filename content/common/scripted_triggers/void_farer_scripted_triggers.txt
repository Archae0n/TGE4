is_void_farer = {
	is_country_type = void_farer
}

is_organic_void_farer = {
	AND = {
		is_country_type = void_farer
		OR = {
			has_origin = origin_void_nomads
			has_origin = origin_void_marauders
			has_origin = origin_void_reavers
		}
	}
}

is_void_nomads = {
	has_origin = origin_void_nomads
}

	OR = {

		has_origin = origin_void_reavers
	}
