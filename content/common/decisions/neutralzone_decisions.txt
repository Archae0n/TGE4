decision_neutral_zoning = {
	resources = {
		category = decisions
		cost = {
			influence = 50
		}
	}
	
	potential = {
		exists = space_owner
		space_owner = { 
			is_same_value = from
		}
		is_star = yes
		from = {
			is_subject = no
			is_ai = no		
		}
		solar_system = {
			any_neighbor_system = {
				or = {
					has_star_flag = sf_neutral_zone
					and = {
						exists = space_owner
						space_owner = { 
							not = { is_same_value = from }
						}
					}
				}
			}
		}
	}
	
	allow = {
		not = {
			solar_system = {
				any_system_planet = {
					is_colony = yes
				}
			}
		}
		not = {
			solar_system = {
				any_ship_in_system = {
					is_ship_class = shipclass_observation_station
				}
			}
		}		
	}
	
	effect = {
		planet_event = { id = neutralzoneevents.1 }
	}
	
	ai_weight = {
		weight = 0
		factor = 0
	}
}

decision_end_neutral_zoning = {	
	resources = {
		category = decisions
		cost = {
			influence = 25
		}
	}
	
	potential = {
		is_star = yes
	}
	
	allow = {
		from = {
			is_ai = no
		}
		solar_system = {
			any_neighbor_system = {
				has_star_flag = sf_neutral_zone
			}
		}		
	}
	
	effect = {
		hidden_effect = {
			from = {
				save_event_target_as = disbandingcountry
			}
			solar_system = {
				every_neighbor_system = {
					limit = { 
						has_star_flag = sf_neutral_zone 
					}
					every_fleet_in_system = {
						limit = {
							or = {
								is_ship_class = shipclass_starbase
								is_ship_class = shipclass_military_station
								is_ship_class = shipclass_research_station
								is_ship_class = shipclass_mining_station
							}
						}
						set_controller = event_target:disbandingcountry
						set_owner = event_target:disbandingcountry
					}
					remove_star_flag = sf_neutral_zone
				}
			}
		}
	}
	
	ai_weight = {
		weight = 0
		factor = 0
	}
}
