raiding = {
	trigger = {
		exists = owner
		# Have ability to raid
		owner = {
			is_raider = yes
			# Is independent OR subject of a raider or fallen empire
			OR = {
				is_subject = no
				overlord = {
					OR = {
						is_raider = yes
						is_fallen_empire = yes
					}
				}
			}
		}
	}

	icon_frame = 4
	default = no

	stop_when_armies_dead = no
	abduct_pops = yes

	planet_damage = 1.25
	army_damage = 0.5

	kill_pop_chance = {
		base = 1.0
	}
	min_pops_to_kill_pop = 2

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 8
		modifier = {
			factor = 0
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
			}
		}
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = { has_claim = root.solar_system }
			}
		}
	}
}
