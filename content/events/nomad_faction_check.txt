namespace = nomad

# Destroy Nomads if no planets, systems and no ships remain
# On yearly pulse
event = {
	id = nomad.208
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		any_country = {
			has_country_flag = is_nomads
		}
	}

	immediate = {
		if = {
			limit = {
				any_country = {
					has_country_flag = is_nomads
					num_ships < 2
					NOT = {
						any_system = {
							any_system_planet = {
								exists = controller
								controller = { has_country_flag = is_nomads }
							}
						}
					}
					NOT = {
						any_system = {
							any_system_planet = {
								exists = owner
								owner = { has_country_flag = is_nomads }
							}
						}
					}
					NOT = {
						any_system = {
							exists = owner
							owner = { has_country_flag = is_nomads }
						}
					}
				}
			}
			random_country = {
				limit = {
					has_country_flag = is_nomads
				}
				destroy_country = yes
			}
		}
	}
}
