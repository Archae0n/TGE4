namespace = aiDomestic


#WHEN AI EMPIRE GOES DOMESTIC, RANDOM PICK DOMESTIC ACTIVITY
country_event = {
	id = aiDomestic.1
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		random_list = {
			10 = { #BOOST REPUTATION
				country_event = {
					id = aiDomestic.2
				}
			}
			30 = { #SWING ETHICS
				modifier = {
					factor = 0
					NOT = {
						has_technology = tech_spy_socialPropaganda
					}
				}
				country_event = {
					id = aiDomestic.3
				}
			}
			30 = { #STOP CRIME
				country_event = {
					id = aiDomestic.4
				}
			}
		}
	}
}


country_event = { #BOOST REPUTATION
	id = aiDomestic.2
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		random_list = {
			60 = {
				every_neighbor_country = {
					limit = {
						is_country_type = default
						not = {
							has_ethic = ethic_gestalt_consciousness
						  }
						has_communications = event_target:AIAgency
					}
					add_opinion_modifier = {
						who = event_target:AIAgency
						modifier = opinion_boostRep_low
					}
				}
			}
			40 = {
				every_neighbor_country = {
					limit = {
						is_country_type = default
						not = {
							has_ethic = ethic_gestalt_consciousness
						  }
						has_communications = event_target:AIAgency
					}
					add_opinion_modifier = {
						who = event_target:AIAgency
						modifier = opinion_boostRep_high
					}
				}
			}
		}
	}
}
