country_event = {
	id = ottereditor.4
	title = ottereditor.1.name
	desc = ottereditor.4.desc
	picture = GFX_evt_star_chart
	is_triggered_only = yes	

	option = {
		name = ottereditor.4.a
		
		allow = {
		    event_target:ott_target = {
		        OR = {
		            AND = {
		                is_moon = no
		                planet_size < 100
		            }
		            planet_size < 50
		        }
		    }
