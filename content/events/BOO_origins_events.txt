# Antediluvian Knowledge Origin - give administrative ai tech and minor artifacts

	id = boo_origins.10
	hide_window = yes

		owner = {
			has_origin = BOO_origin_antediluvian_knowledge
		}
	}

	immediate = {
		give_technology = {
			tech = tech_administrative_ai
			message = no
		}

	}
}
